$$\Delta_n(\F, \sigma^2) \ge \dfrac{C_3}{ n^{1/4}}, \text{ when } p=1, q=2.$$
 $$\Delta_n(\F, \sigma^2) \ge \dfrac{C_4}{ n^{1/3}},  \text{ when } p=q=2.$$
\end{theorem}
